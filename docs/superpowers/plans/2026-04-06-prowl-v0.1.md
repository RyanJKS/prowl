# Prowl v0.1 — Functional Core Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working TUI that searches directories via fuzzy matching, shows an overview preview, and dispatches actions (open in editor, cd, yank path) — the minimum viable Prowl.

**Architecture:** Synchronous main thread (crossterm poll → process input → nucleo tick → ratatui draw) communicating with a tokio background runtime via `std::sync::mpsc`. Nucleo manages its own scoring threads. A `Vec<PathEntry>` candidate cache alongside nucleo enables safe re-traversal. Vim-style modal input (insert/normal) with configurable action keybindings.

**Tech Stack:** Rust, ratatui, crossterm, ignore, nucleo, tokio, toml/serde, arboard, camino, anyhow, clap

```rust
            let contents = std::fs::read_to_string(&config_path)
                .with_context(|| format!("Failed to read config file: {config_path}"))?;

            .iter()
            .map(|r| Self::expand_path(r))
            .filter(|p| p.exists())
            .collect()
    }

        Utf8PathBuf::from(format!("{home}/.config/prowl/config.toml"))
    } else {
        Utf8PathBuf::from("prowl/config.toml")
    }
```

```rust
#[derive(Debug, Default, PartialEq)]
pub struct ParsedQuery {
    /// The fuzzy portion passed to nucleo.
    pub fuzzy: String,
    /// Tag filters extracted from #tag tokens.
    pub tags: Vec<String>,
    /// Negation filters extracted from !word tokens.
    pub negations: Vec<String>,
    /// Path prefix filter extracted from ^path token (only one allowed).
    pub prefix: Option<String>,
    /// Exact substring matches extracted from 'word tokens.
    pub exact: Vec<String>,
}

/// Parse a raw query string into structured components.
///
/// Tokens:
/// - `#word` → tag filter
/// - `!word` → negation filter
/// - `^path` → prefix filter (first occurrence only)
/// - `'word` → exact substring match
/// - `\#`, `\!`, `\^`, `\'` → escaped literals (treated as fuzzy)
/// - everything else → fuzzy query
pub fn parse_query(input: &str) -> ParsedQuery {
    let mut query = ParsedQuery::default();
    let mut fuzzy_parts: Vec<String> = Vec::new();

    for token in input.split_whitespace() {
        if token.starts_with("\\#") || token.starts_with("\\!") || token.starts_with("\\^") || token.starts_with("\\'") {
            // Escaped special character — strip the backslash, treat as fuzzy
            fuzzy_parts.push(token[1..].to_string());
        } else if let Some(tag) = token.strip_prefix('#') {
            if !tag.is_empty() {
                query.tags.push(tag.to_string());
            }
        } else if let Some(neg) = token.strip_prefix('!') {
            if !neg.is_empty() {
                query.negations.push(neg.to_string());
            }
        } else if let Some(pfx) = token.strip_prefix('^') {
            if query.prefix.is_none() && !pfx.is_empty() {
                query.prefix = Some(pfx.to_string());
            } else {
                // Already have a prefix or empty — treat as fuzzy
                fuzzy_parts.push(token.to_string());
            }
        } else if let Some(exact) = token.strip_prefix('\'') {
            if !exact.is_empty() {
                query.exact.push(exact.to_string());
            }
        } else {
            fuzzy_parts.push(token.to_string());
        }
    }

    query.fuzzy = fuzzy_parts.join(" ");
    query
}
```

```rust
#[derive(Debug, Clone)]
pub struct PathEntry {

    pub path: Utf8PathBuf,

    pub name: String,

}

/// Walk the given root directories and collect all directories found.
///
/// This is a synchronous function intended to be called from a background thread.
/// Results are collected into a Vec. For v0.1 we use a single root;
/// the Vec<String> parameter supports future multi-root.
pub fn walk_directories(roots: &[String], opts: &WalkOptions) -> Vec<PathEntry> {
    let mut entries = Vec::new();

    for root in roots {
        let root_path = std::path::Path::new(root);
        if !root_path.exists() {
            continue;
        }

        let walker = WalkBuilder::new(root_path)
            .max_depth(Some(opts.max_depth))
            .hidden(!opts.show_hidden)
            .git_ignore(opts.respect_gitignore)
            .git_global(opts.respect_gitignore)
            .git_exclude(opts.respect_gitignore)
            .follow_links(opts.follow_symlinks)
            .build();

        for entry in walker.flatten() {
            // Skip the root directory itself
            if entry.path() == root_path {
                continue;
            }

            // Only collect directories
            if !entry.file_type().map_or(false, |ft| ft.is_dir()) {
                continue;
            }

            let path_str = match entry.path().to_str() {
                Some(s) => s,
                None => continue, // Skip non-UTF8 paths
            };

            let utf8_path = Utf8PathBuf::from(path_str);
            let name = utf8_path
                .file_name()
                .unwrap_or("")
                .to_string();
            let parent_display = utf8_path
                .parent()
                .map(|p| p.to_string())
                .unwrap_or_default();

            entries.push(PathEntry {
                path: utf8_path,
                name,
                parent_display,
            });
        }
    }

    entries
}
```

```rust
use crate::config::{ActionDef, ActionsConfig};
use shell_escape::escape;
use std::borrow::Cow;

#[derive(Debug, Clone, PartialEq)]
pub enum ActionMode {
    Detach,
    Suspend,
    Builtin,
}

impl ActionMode {
    pub fn from_str(s: &str) -> Self {
        match s {
            "suspend" => Self::Suspend,
            "builtin" => Self::Builtin,
            _ => Self::Detach,
        }
    }
}

/// Substitute `{}` and `{*}` in a command string with shell-escaped paths.
///
/// - `{}` is replaced with the first path (shell-escaped).
/// - `{*}` is replaced with all paths joined by spaces (each shell-escaped).
/// - If no placeholder is present, the command is returned as-is (for builtins).
pub fn substitute_path(cmd: &str, paths: &[String]) -> String {
    let mut result = cmd.to_string();

    if result.contains("{*}") {
        let escaped: Vec<String> = paths
            .iter()
            .map(|p| escape(Cow::Borrowed(p.as_str())).to_string())
            .collect();
        result = result.replace("{*}", &escaped.join(" "));
    }

    if result.contains("{}") {
        let first = paths.first().map(|p| p.as_str()).unwrap_or("");
        let escaped = escape(Cow::Borrowed(first)).to_string();
        result = result.replace("{}", &escaped);
    }

    result
}

/// Resolve which action definition applies for a given key press.
///
/// Resolution order (most specific first):
/// 1. `<input_mode>-<tab_mode>` (e.g. "normal-files")
/// 2. `<input_mode>` (e.g. "normal")
/// 3. None
pub fn resolve_action<'a>(
    actions: &'a ActionsConfig,
    input_mode: &str,
    tab_mode: &str,
    key: &str,
) -> Option<&'a ActionDef> {
    // Check mode-tab specific first
    let specific = match (input_mode, tab_mode) {
        ("normal", "files") => actions.normal_files.get(key),
        ("insert", "files") => actions.insert_files.get(key),
        ("normal", "bookmarks") => actions.normal_bookmarks.get(key),
        ("insert", "bookmarks") => actions.insert_bookmarks.get(key),
        _ => None,
    };

    if specific.is_some() {
        return specific;
    }

    // Fall back to base input mode
    match input_mode {
        "normal" => actions.normal.get(key),
        "insert" => actions.insert.get(key),
        _ => None,
    }
}
```

```rust
#[derive(Parser, Debug)]
#[command(name = "prowl", version, about = "A fast TUI for navigating your filesystem")]
struct Cli {
    /// Root directory to search from
    path: Option<String>,

    /// Print shell integration script
    #[arg(long, value_name = "SHELL")]
    init: Option<String>,

    /// Open config file in $EDITOR
    #[arg(long)]
    config: bool,

    /// Print version with git hash
    #[arg(long = "long-version", hide = true)]
    long_version: bool,
}

fn main() -> Result<()> {
    let cli = Cli::parse();

    // Handle --init
    if let Some(shell_name) = &cli.init {
        print!("{}", shell::init_script(shell_name));
        return Ok(());
    }

    // Handle --config
    if cli.config {
        let config_path = prowl::config::config_path();
        let editor = std::env::var("EDITOR").unwrap_or_else(|_| "vi".to_string());

        // Create config directory if needed
        if let Some(parent) = config_path.parent() {
            std::fs::create_dir_all(parent.as_std_path())?;
        }

        // Create default config if it doesn't exist
        if !config_path.exists() {
            std::fs::write(&config_path, include_str!("default_config.toml"))?;
        }

        std::process::Command::new(&editor)
            .arg(config_path.as_str())
            .status()?;
        return Ok(());
    }

    // Load config
    let config = Config::load().unwrap_or_else(|e| {
        eprintln!("Warning: failed to load config: {e}. Using defaults.");
        Config::default()
    });

    // Determine root path
    let root = match &cli.path {
        Some(p) => Config::expand_path(p).to_string(),
        None => {
            if config.roots.is_empty() {
                std::env::current_dir()?
                    .to_str()
                    .unwrap_or(".")
                    .to_string()
            } else {
                // For v0.1, use the first root. Multi-root comes in v0.5.
                Config::expand_path(&config.roots[0]).to_string()
            }
        }
    };

    println!("prowl v{} ({}) — root: {root}", env!("CARGO_PKG_VERSION"), env!("PROWL_GIT_HASH"));
    println!("TUI launch will be implemented in Task 8.");

    Ok(())
}
```

Write `src/app.rs`:

```rust
use crate::actions::{resolve_action, substitute_path, ActionMode};
use crate::config::Config;
use crate::search::query::parse_query;
use crate::search::walker::{walk_directories, PathEntry, WalkOptions};
use anyhow::Result;
use camino::Utf8PathBuf;
use crossterm::event::{self, Event, KeyCode, KeyEvent, KeyModifiers};
use nucleo::{Config as NucleoConfig, Nucleo, pattern::{CaseMatching, Normalization, Pattern}};
use std::sync::mpsc;
use std::time::Duration;

/// Events sent from background threads to the main thread.
pub enum AppEvent {
    /// Walker finished — here are all the directory entries found.
    WalkComplete(Vec<PathEntry>),
    /// Preview content ready.
    PreviewReady(String),
}

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum InputMode {
    Insert,
    Normal,
}

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum TabMode {
    Dirs,
}

pub struct App {
    pub config: Config,
    pub should_quit: bool,
    pub input_mode: InputMode,
    pub tab_mode: TabMode,
    pub query: String,
    pub cursor_pos: usize,
    pub selected_index: usize,
    pub preview_scroll: u16,
    pub root: String,

    // Search state
    pub candidates: Vec<PathEntry>,
    pub filtered_results: Vec<PathEntry>,
    pub nucleo: Nucleo<String>,
    pub is_walking: bool,

    // Preview state
    pub preview_content: Vec<String>,

    // Channels
    pub event_rx: mpsc::Receiver<AppEvent>,
    pub event_tx: mpsc::Sender<AppEvent>,

    // Tokio runtime handle
    pub rt: tokio::runtime::Handle,

    // Result of cd action (path to write to lastdir)
    pub cd_path: Option<String>,
}

impl App {
    pub fn new(config: Config, root: String) -> Self {
        let (event_tx, event_rx) = mpsc::channel();

        // Spawn a tokio runtime on a background thread
        let rt = tokio::runtime::Builder::new_multi_thread()
            .worker_threads(2)
            .enable_all()
            .build()
            .expect("Failed to create tokio runtime");
        let rt_handle = rt.handle().clone();

        // Keep the runtime alive by leaking it (it lives for the entire process)
        std::mem::forget(rt);

        let nucleo = Nucleo::new(
            NucleoConfig::DEFAULT,
            std::sync::Arc::new(|| {}),
            None,
            1, // one column (the path string)
        );

        let mut app = Self {
            config,
            should_quit: false,
            input_mode: InputMode::Insert,
            tab_mode: TabMode::Dirs,
            query: String::new(),
            cursor_pos: 0,
            selected_index: 0,
            preview_scroll: 0,
            root: root.clone(),
            candidates: Vec::new(),
            filtered_results: Vec::new(),
            nucleo,
            is_walking: true,
            preview_content: vec!["Loading...".to_string()],
            event_rx,
            event_tx,
            rt: rt_handle,
            cd_path: None,
        };

        // Kick off initial directory walk
        app.start_walk();
        app
    }

    fn start_walk(&self) {
        let tx = self.event_tx.clone();
        let roots = vec![self.root.clone()];
        let opts = WalkOptions {
            max_depth: self.config.default_depth,
            show_hidden: self.config.show_hidden,
            respect_gitignore: self.config.respect_gitignore,
            follow_symlinks: self.config.follow_symlinks,
        };

        self.rt.spawn(async move {
            let entries = tokio::task::spawn_blocking(move || {
                walk_directories(&roots, &opts)
            })
            .await
            .unwrap_or_default();

            let _ = tx.send(AppEvent::WalkComplete(entries));
        });
    }

    /// Process all pending background events.
    pub fn process_events(&mut self) {
        while let Ok(event) = self.event_rx.try_recv() {
            match event {
                AppEvent::WalkComplete(entries) => {
                    self.candidates = entries;
                    self.is_walking = false;
                    self.rebuild_nucleo();
                    self.update_filtered_results();
                    self.update_preview();
                }
                AppEvent::PreviewReady(content) => {
                    self.preview_content = content.lines().map(String::from).collect();
                }
            }
        }
    }

    fn rebuild_nucleo(&mut self) {
        self.nucleo = Nucleo::new(
            NucleoConfig::DEFAULT,
            std::sync::Arc::new(|| {}),
            None,
            1,
        );

        let injector = self.nucleo.injector();
        for entry in &self.candidates {
            let _ = injector.push(entry.path.as_str().to_string(), |s, cols| {
                cols[0] = s.as_str().into();
            });
        }
    }

    fn update_filtered_results(&mut self) {
        let parsed = parse_query(&self.query);

        // Update nucleo pattern with fuzzy portion
        if !parsed.fuzzy.is_empty() {
            self.nucleo.pattern.reparse(
                0,
                &parsed.fuzzy,
                CaseMatching::Smart,
                Normalization::Smart,
                false,
            );
        } else {
            self.nucleo.pattern.reparse(
                0,
                "",
                CaseMatching::Smart,
                Normalization::Smart,
                false,
            );
        }

        // Tick nucleo to process
        self.nucleo.tick(10);

        let snapshot = self.nucleo.snapshot();

        // Build a path-to-entry lookup
        let entry_map: std::collections::HashMap<&str, &PathEntry> = self
            .candidates
            .iter()
            .map(|e| (e.path.as_str(), e))
            .collect();

        self.filtered_results.clear();

        for idx in 0..snapshot.matched_item_count().min(self.config.max_results as u32) {
            let item = &snapshot.get_matched_item(idx);
            let path_str: String = item.data.to_string();

            if let Some(entry) = entry_map.get(path_str.as_str()) {
                let entry = (*entry).clone();

                // Apply post-filters
                let path_lower = entry.path.as_str().to_lowercase();

                // Negation filter
                if parsed.negations.iter().any(|n| path_lower.contains(&n.to_lowercase())) {
                    continue;
                }

                // Prefix filter
                if let Some(ref prefix) = parsed.prefix {
                    let expanded = Config::expand_path(prefix);
                    if !entry.path.as_str().starts_with(expanded.as_str()) {
                        continue;
                    }
                }

                // Exact match filter
                if !parsed.exact.is_empty() {
                    let all_match = parsed.exact.iter().all(|e| path_lower.contains(&e.to_lowercase()));
                    if !all_match {
                        continue;
                    }
                }

                self.filtered_results.push(entry);
            }
        }

        // Clamp selected index
        if !self.filtered_results.is_empty() {
            self.selected_index = self.selected_index.min(self.filtered_results.len() - 1);
        } else {
            self.selected_index = 0;
        }
    }

    fn update_preview(&mut self) {
        if let Some(entry) = self.filtered_results.get(self.selected_index) {
            let path = entry.path.clone();
            let tx = self.event_tx.clone();

            self.rt.spawn(async move {
                let content = tokio::task::spawn_blocking(move || {
                    generate_overview_preview(&path)
                })
                .await
                .unwrap_or_else(|_| "Preview error".to_string());

                let _ = tx.send(AppEvent::PreviewReady(content));
            });
        } else {
            self.preview_content = vec!["No results".to_string()];
        }
    }

    /// Handle a crossterm key event.
    pub fn handle_key(&mut self, key: KeyEvent) {
        // Reserved keys that work in both modes
        match (key.code, key.modifiers) {
            (KeyCode::Char('c'), KeyModifiers::CONTROL) => {
                self.should_quit = true;
                return;
            }
            (KeyCode::Up, KeyModifiers::NONE) | (KeyCode::Char('k'), KeyModifiers::CONTROL) => {
                self.move_selection(-1);
                return;
            }
            (KeyCode::Down, KeyModifiers::NONE) | (KeyCode::Char('j'), KeyModifiers::CONTROL) => {
                self.move_selection(1);
                return;
            }
            (KeyCode::PageUp, _) | (KeyCode::Char('u'), KeyModifiers::CONTROL) => {
                self.move_selection(-10);
                return;
            }
            (KeyCode::PageDown, _) | (KeyCode::Char('d'), KeyModifiers::CONTROL) => {
                self.move_selection(10);
                return;
            }
            (KeyCode::Up, KeyModifiers::SHIFT) => {
                self.preview_scroll = self.preview_scroll.saturating_sub(1);
                return;
            }
            (KeyCode::Down, KeyModifiers::SHIFT) => {
                self.preview_scroll = self.preview_scroll.saturating_add(1);
                return;
            }
            _ => {}
        }

        match self.input_mode {
            InputMode::Insert => self.handle_insert_key(key),
            InputMode::Normal => self.handle_normal_key(key),
        }
    }

    fn handle_insert_key(&mut self, key: KeyEvent) {
        match (key.code, key.modifiers) {
            (KeyCode::Esc, _) => {
                if !self.query.is_empty() {
                    self.query.clear();
                    self.cursor_pos = 0;
                    self.update_filtered_results();
                    self.update_preview();
                } else {
                    self.input_mode = InputMode::Normal;
                }
            }
            (KeyCode::Enter, KeyModifiers::NONE) => {
                self.dispatch_action("insert", "enter");
            }
            (KeyCode::Backspace, _) => {
                if self.cursor_pos > 0 {
                    self.query.remove(self.cursor_pos - 1);
                    self.cursor_pos -= 1;
                    self.update_filtered_results();
                    self.update_preview();
                }
            }
            (KeyCode::Char(c), KeyModifiers::CONTROL) => {
                // Check for insert-mode ctrl actions
                let key_name = format!("ctrl_{c}");
                self.dispatch_action("insert", &key_name);
            }
            (KeyCode::Char(c), KeyModifiers::NONE | KeyModifiers::SHIFT) => {
                self.query.insert(self.cursor_pos, c);
                self.cursor_pos += 1;
                self.update_filtered_results();
                self.update_preview();
            }
            _ => {}
        }
    }

    fn handle_normal_key(&mut self, key: KeyEvent) {
        match (key.code, key.modifiers) {
            (KeyCode::Esc, _) | (KeyCode::Char('q'), KeyModifiers::NONE) => {
                self.should_quit = true;
            }
            (KeyCode::Char('/'), KeyModifiers::NONE) | (KeyCode::Char('i'), KeyModifiers::NONE) => {
                self.input_mode = InputMode::Insert;
            }
            (KeyCode::Char('j'), KeyModifiers::NONE) => {
                self.move_selection(1);
            }
            (KeyCode::Char('k'), KeyModifiers::NONE) => {
                self.move_selection(-1);
            }
            (KeyCode::Enter, KeyModifiers::NONE) => {
                self.dispatch_action("normal", "enter");
            }
            (KeyCode::Char(c), KeyModifiers::NONE) => {
                let key_name = c.to_string();
                self.dispatch_action("normal", &key_name);
            }
            (KeyCode::Char(c), KeyModifiers::CONTROL) => {
                let key_name = format!("ctrl_{c}");
                self.dispatch_action("normal", &key_name);
            }
            _ => {}
        }
    }

    fn dispatch_action(&mut self, input_mode: &str, key: &str) {
        let tab_str = match self.tab_mode {
            TabMode::Dirs => "dirs",
        };

        let action = resolve_action(&self.config.actions, input_mode, tab_str, key);
        let action = match action {
            Some(a) => a.clone(),
            None => return,
        };

        let selected_path = match self.filtered_results.get(self.selected_index) {
            Some(entry) => entry.path.to_string(),
            None => return,
        };

        let paths = vec![selected_path.clone()];
        let mode = ActionMode::from_str(&action.mode);

        match mode {
            ActionMode::Builtin => {
                match action.cmd.as_str() {
                    "__prowl_cd" => {
                        self.cd_path = Some(selected_path);
                        self.should_quit = true;
                    }
                    "__prowl_yank_abs" => {
                        if let Ok(mut clipboard) = arboard::Clipboard::new() {
                            let _ = clipboard.set_text(&selected_path);
                        }
                    }
                    "__prowl_yank_rel" => {
                        let home = std::env::var("HOME").unwrap_or_default();
                        let rel = selected_path
                            .strip_prefix(&home)
                            .map(|r| format!("~{r}"))
                            .unwrap_or(selected_path);
                        if let Ok(mut clipboard) = arboard::Clipboard::new() {
                            let _ = clipboard.set_text(&rel);
                        }
                    }
                    _ => {}
                }
            }
            ActionMode::Detach => {
                let cmd = substitute_path(&action.cmd, &paths);
                let _ = std::process::Command::new("sh")
                    .arg("-c")
                    .arg(&cmd)
                    .stdin(std::process::Stdio::null())
                    .stdout(std::process::Stdio::null())
                    .stderr(std::process::Stdio::null())
                    .spawn();
            }
            ActionMode::Suspend => {
                self.suspend_for_command(&action.cmd, &paths);
            }
        }
    }

    fn suspend_for_command(&mut self, cmd: &str, paths: &[String]) {
        use crossterm::{execute, terminal};

        let cmd = substitute_path(cmd, paths);

        // Leave TUI
        let _ = execute!(std::io::stdout(), crossterm::terminal::LeaveAlternateScreen);
        let _ = terminal::disable_raw_mode();

        // Run command
        let _ = std::process::Command::new("sh")
            .arg("-c")
            .arg(&cmd)
            .status();

        // Re-enter TUI
        let _ = terminal::enable_raw_mode();
        let _ = execute!(std::io::stdout(), crossterm::terminal::EnterAlternateScreen);
    }

    fn move_selection(&mut self, delta: i32) {
        let len = self.filtered_results.len();
        if len == 0 {
            return;
        }

        let new_idx = if delta < 0 {
            self.selected_index.saturating_sub(delta.unsigned_abs() as usize)
        } else {
            (self.selected_index + delta as usize).min(len - 1)
        };

        if new_idx != self.selected_index {
            self.selected_index = new_idx;
            self.preview_scroll = 0;
            self.update_preview();
        }
    }
}

/// Generate a plain-text overview preview for a directory.
fn generate_overview_preview(path: &Utf8PathBuf) -> String {
    let mut lines = Vec::new();
    let std_path = path.as_std_path();

    // Directory name header
    lines.push(format!("  {}", path.file_name().unwrap_or(path.as_str())));
    lines.push(format!("  {}", path));
    lines.push(String::new());

    // 2-level directory tree
    lines.push("  Tree:".to_string());
    if let Ok(entries) = std::fs::read_dir(std_path) {
        let mut entries: Vec<_> = entries
            .flatten()
            .filter(|e| {
                e.file_name()
                    .to_str()
                    .map_or(false, |n| !n.starts_with('.'))
            })
            .collect();
        entries.sort_by_key(|e| e.file_name());

        for (i, entry) in entries.iter().enumerate().take(20) {
            let name = entry.file_name();
            let name = name.to_string_lossy();
            let is_dir = entry.file_type().map_or(false, |ft| ft.is_dir());
            let prefix = if i == entries.len().min(20) - 1 { "  └── " } else { "  ├── " };
            let suffix = if is_dir { "/" } else { "" };
            lines.push(format!("{prefix}{name}{suffix}"));

            // Show one level deeper for directories
            if is_dir {
                if let Ok(sub_entries) = std::fs::read_dir(entry.path()) {
                    let mut sub: Vec<_> = sub_entries
                        .flatten()
                        .filter(|e| {
                            e.file_name()
                                .to_str()
                                .map_or(false, |n| !n.starts_with('.'))
                        })
                        .collect();
                    sub.sort_by_key(|e| e.file_name());

                    let tree_prefix = if i == entries.len().min(20) - 1 { "      " } else { "  │   " };
                    for (j, sub_entry) in sub.iter().enumerate().take(5) {
                        let sub_name = sub_entry.file_name();
                        let sub_name = sub_name.to_string_lossy();
                        let sub_is_dir = sub_entry.file_type().map_or(false, |ft| ft.is_dir());
                        let sub_prefix = if j == sub.len().min(5) - 1 { "└── " } else { "├── " };
                        let sub_suffix = if sub_is_dir { "/" } else { "" };
                        lines.push(format!("{tree_prefix}{sub_prefix}{sub_name}{sub_suffix}"));
                    }
                    if sub.len() > 5 {
                        lines.push(format!("{tree_prefix}… {} more", sub.len() - 5));
                    }
                }
            }
        }
        if entries.len() > 20 {
            lines.push(format!("  … {} more entries", entries.len() - 20));
        }
    }

    lines.push(String::new());

    // README snippet (first 6 lines)
    let readme_names = ["README.md", "README.txt", "README", "readme.md"];
    for readme_name in &readme_names {
        let readme_path = std_path.join(readme_name);
        if readme_path.exists() {
            if let Ok(content) = std::fs::read_to_string(&readme_path) {
                lines.push(format!("  README ({readme_name}):"));
                for line in content.lines().take(6) {
                    lines.push(format!("  {line}"));
                }
                if content.lines().count() > 6 {
                    lines.push("  …".to_string());
                }
            }
            break;
        }
    }

    lines.join("\n")
}
```

```rust
pub fn run(config: Config, root: String) -> Result<()> {
    use crossterm::{execute, terminal};

    // Setup terminal
    terminal::enable_raw_mode()?;
    let mut stdout = std::io::stdout();
    execute!(stdout, terminal::EnterAlternateScreen)?;
    let backend = ratatui::backend::CrosstermBackend::new(stdout);
    let mut terminal = ratatui::Terminal::new(backend)?;

    let mut app = App::new(config, root);

    // Main event loop
    loop {
        // Process background events
        app.process_events();

        // Tick nucleo
        app.nucleo.tick(10);

        // Draw UI
        terminal.draw(|f| crate::ui::draw(f, &app))?;

        // Poll for input (16ms timeout ≈ 60fps)
        if event::poll(Duration::from_millis(16))? {
            if let Event::Key(key) = event::read()? {
                app.handle_key(key);
            }
        }

        if app.should_quit {
            break;
        }
    }

    // Restore terminal
    crossterm::terminal::disable_raw_mode()?;
    execute!(
        terminal.backend_mut(),
        terminal::LeaveAlternateScreen
    )?;

    // Handle cd action — write lastdir file
    if let Some(cd_path) = &app.cd_path {
        let lastdir = Config::expand_path(&app.config.paths.lastdir_file);
        if let Some(parent) = lastdir.parent() {
            let _ = std::fs::create_dir_all(parent.as_std_path());
        }
        std::fs::write(lastdir.as_std_path(), cd_path)?;
    }

    Ok(())
}
```

```rust
use ratatui::layout::{Constraint, Direction, Layout, Rect};

pub struct AppLayout {
    pub search: Rect,
    pub results: Rect,
    pub preview: Rect,
    pub keybind_bar: Rect,
}

pub fn compute_layout(area: Rect) -> AppLayout {
    // Vertical split: search bar (3 lines) | main content | keybind bar (1 line)
    let vertical = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(3),  // search bar
            Constraint::Min(5),    // main content
            Constraint::Length(1), // keybind bar
        ])
        .split(area);

    let search = vertical[0];
    let main_area = vertical[1];
    let keybind_bar = vertical[2];

    // Horizontal split: results (50%) | preview (50%)
    // Collapse preview below 100 columns
    let show_preview = area.width >= 100;

    let (results, preview) = if show_preview {
        let horizontal = Layout::default()
            .direction(Direction::Horizontal)
            .constraints([Constraint::Percentage(50), Constraint::Percentage(50)])
            .split(main_area);
        (horizontal[0], horizontal[1])
    } else {
        (main_area, Rect::default())
    };

    AppLayout {
        search,
        results,
        preview,
        keybind_bar,
    }
}
```
